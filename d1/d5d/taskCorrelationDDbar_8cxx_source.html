123  void process(aod::DDbarPairFull const& pairEntries)
124  {
125  for (auto& pairEntry : pairEntries) {
126  //define variables for widely used quantities
127  double deltaPhi = pairEntry.deltaPhi();
128  double deltaEta = pairEntry.deltaEta();
129  double ptD = pairEntry.ptD();
130  double ptDbar = pairEntry.ptDbar();
131  double massD = pairEntry.mD();
132  double massDbar = pairEntry.mDbar();
133 
134  //fill 2D invariant mass plots
135  registry.fill(HIST("hMass2DCorrelationPairs"), massD, massDbar, ptD, ptDbar);
136 
137  //reject entries outside pT ranges of interest
138  int pTBinD = o2::analysis::findBin(binsCorrelations, ptD);
139  int pTBinDbar = o2::analysis::findBin(binsCorrelations, ptDbar);
140  if (pTBinD == -1 || pTBinDbar == -1) { //at least one particle outside accepted pT range
141  continue;
142  }
143 
144  //check if correlation entry belongs to signal region, sidebands or is outside both, and fill correlation plots
145  if (massD > signalRegionInner->at(pTBinD) && massD < signalRegionOuter->at(pTBinD) && massDbar > signalRegionInner->at(pTBinDbar) && massDbar < signalRegionOuter->at(pTBinDbar)) {
146  //in signal region
147  registry.fill(HIST("hCorrel2DVsPtSignalRegion"), deltaPhi, deltaEta, ptD, ptDbar);
148  registry.fill(HIST("hCorrel2DPtIntSignalRegion"), deltaPhi, deltaEta);
149  registry.fill(HIST("hDeltaEtaPtIntSignalRegion"), deltaEta);
150  registry.fill(HIST("hDeltaPhiPtIntSignalRegion"), deltaPhi);
151  registry.fill(HIST("hDeltaPtDDbarSignalRegion"), ptDbar - ptD);
152  registry.fill(HIST("hDeltaPtMaxMinSignalRegion"), std::abs(ptDbar - ptD));
153  }
154 
155  if ((massD > sidebandLeftInner->at(pTBinD) && massD < sidebandLeftOuter->at(pTBinD) && massDbar > sidebandLeftInner->at(pTBinDbar) && massDbar < sidebandRightOuter->at(pTBinDbar)) ||
156  (massD > sidebandRightInner->at(pTBinD) && massD < sidebandRightOuter->at(pTBinD) && massDbar > sidebandLeftInner->at(pTBinDbar) && massDbar < sidebandRightOuter->at(pTBinDbar)) ||
157  (massD > sidebandLeftInner->at(pTBinD) && massD < sidebandRightOuter->at(pTBinD) && massDbar > sidebandLeftInner->at(pTBinDbar) && massDbar < sidebandLeftOuter->at(pTBinDbar)) ||
158  (massD > sidebandLeftInner->at(pTBinD) && massD < sidebandRightOuter->at(pTBinD) && massDbar > sidebandRightInner->at(pTBinDbar) && massDbar < sidebandRightOuter->at(pTBinDbar))) {
159  //in sideband region
160  registry.fill(HIST("hCorrel2DVsPtSidebands"), deltaPhi, deltaEta, ptD, ptDbar);
161  registry.fill(HIST("hCorrel2DPtIntSidebands"), deltaPhi, deltaEta);
162  registry.fill(HIST("hDeltaEtaPtIntSidebands"), deltaEta);
163  registry.fill(HIST("hDeltaPhiPtIntSidebands"), deltaPhi);
164  registry.fill(HIST("hDeltaPtDDbarSidebands"), ptDbar - ptD);
165  registry.fill(HIST("hDeltaPtMaxMinSidebands"), std::abs(ptDbar - ptD));
166  }
167  } //end loop
168  }
169 };

233  void process(aod::DDbarPairFull const& pairEntries)
234  {
235  for (auto& pairEntry : pairEntries) {
236  //define variables for widely used quantities
237  double deltaPhi = pairEntry.deltaPhi();
238  double deltaEta = pairEntry.deltaEta();
239  double ptD = pairEntry.ptD();
240  double ptDbar = pairEntry.ptDbar();
241  double massD = pairEntry.mD();
242  double massDbar = pairEntry.mDbar();
243 
244  //reject entries outside pT ranges of interest
245  int pTBinD = o2::analysis::findBin(binsCorrelations, ptD);
246  int pTBinDbar = o2::analysis::findBin(binsCorrelations, ptDbar);
247  if (pTBinD == -1 || pTBinDbar == -1) { //at least one particle outside accepted pT range
248  continue;
249  }
250 
251  //fill 2D invariant mass plots
252  switch (pairEntry.signalStatus()) {
253  case 0: //D Bkg, Dbar Bkg
254  registry.fill(HIST("hMass2DCorrelationPairsMCRecBkgBkg"), massD, massDbar, ptD, ptDbar);
255  break;
256  case 1: //D Bkg, Dbar Sig
257  registry.fill(HIST("hMass2DCorrelationPairsMCRecBkgSig"), massD, massDbar, ptD, ptDbar);
258  break;
259  case 2: //D Sig, Dbar Bkg
260  registry.fill(HIST("hMass2DCorrelationPairsMCRecSigBkg"), massD, massDbar, ptD, ptDbar);
261  break;
262  case 3: //D Sig, Dbar Sig
263  registry.fill(HIST("hMass2DCorrelationPairsMCRecSigSig"), massD, massDbar, ptD, ptDbar);
264  break;
265  default: //should not happen for MC reco
266  break;
267  }
268 
269  //check if correlation entry belongs to signal region, sidebands or is outside both, and fill correlation plots
