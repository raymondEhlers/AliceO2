32 struct JetFinderTaskBase {
33  Produces<o2::aod::Jets> jetsTable;
34  Produces<o2::aod::JetConstituents> constituentsTable;
35  Produces<o2::aod::JetConstituentsSub> constituentsSubTable;
36  OutputObj<TH1F> hJetPt{"h_jet_pt"};
37  OutputObj<TH1F> hJetPhi{"h_jet_phi"};
38  OutputObj<TH1F> hJetEta{"h_jet_eta"};
39  OutputObj<TH1F> hJetN{"h_jet_n"};
40 
41  Configurable<float> vertexZCut{"vertexZCut", 10.0f, "Accepted z-vertex range"};
42  Configurable<float> trackPtCut{"trackPtCut", 0.1, "minimum constituent pT"};
43  Configurable<float> trackEtaCut{"trackEtaCut", 0.9, "constituent eta cut"};
44  Configurable<bool> DoRhoAreaSub{"DoRhoAreaSub", false, "do rho area subtraction"};
45  Configurable<bool> DoConstSub{"DoConstSub", false, "do constituent subtraction"};
46  Configurable<float> jetPtMin{"jetPtMin", 10.0, "minimum jet pT"};
47  Configurable<float> jetR{"jetR", 0.4, "jet resolution"};
48 
49  Filter collisionFilter = nabs(aod::collision::posZ) < vertexZCut;
50  Filter trackFilter = (nabs(aod::track::eta) < trackEtaCut) && (aod::track::isGlobalTrack == (uint8_t) true) && (aod::track::pt > trackPtCut);
51 
52  std::vector<fastjet::PseudoJet> jets;
53  std::vector<fastjet::PseudoJet> inputParticles;
54  JetFinder jetFinder; //should be a configurable but for now this cant be changed on hyperloop
55 
56  void init(InitContext const&)
57  {
58  hJetPt.setObject(new TH1F("h_jet_pt", "jet p_{T};p_{T} (GeV/#it{c})",
59  100, 0., 100.));
60  hJetPhi.setObject(new TH1F("h_jet_phi", "jet #phi;#phi",
61  80, -1., 7.));
62  hJetEta.setObject(new TH1F("h_jet_eta", "jet #eta;#eta",
63  70, -0.7, 0.7));
64  hJetN.setObject(new TH1F("h_jet_n", "jet n;n constituents",
65  30, 0., 30.));
66  if (DoRhoAreaSub) {
67  jetFinder.setBkgSubMode(JetFinder::BkgSubMode::rhoAreaSub);
68  }
69  if (DoConstSub) {
70  jetFinder.setBkgSubMode(JetFinder::BkgSubMode::constSub);
71  }
72  jetFinder.jetPtMin = jetPtMin;
73  jetFinder.jetR = jetR;
74  }
75 
76  void processBase()
77  {
78  fastjet::ClusterSequenceArea clusterSeq(jetFinder.findJets(inputParticles, jets));
79 
80  for (const auto& jet : jets) {
81  jetsTable(collision, jet.pt(), jet.eta(), jet.phi(),
82  jet.E(), jet.m(), jet.area(), -1);
83  hJetPt->Fill(jet.pt());
84  hJetPhi->Fill(jet.phi());
85  hJetEta->Fill(jet.eta());
86  hJetN->Fill(jet.constituents().size());
87  for (const auto& constituent : jet.constituents()) { //event or jetwise
88  if (DoConstSub) {
89  constituentsSubTable(jetsTable.lastIndex(), constituent.pt(), constituent.eta(), constituent.phi(),
90  constituent.E(), constituent.m());
91  }
92  constituentsTable(jetsTable.lastIndex(), constituent.user_index());
93  }
94  }
95  }
96 
97  void processBegin()
98  {
99  jets.clear();
100  inputParticles.clear();
101  }
102 
103 };

129 struct JetFinderTaskFull : public JetFinderTaskBase {
130  o2::emcal::Geometry* geometry
131  void Init()
132  {
133  // Setup geometry for EMCal
134  // This would be a candidate to be moved elsewhere, but it depends on what is stored in the cluster table.
135  // NOTE: The geometry manager isn't necessary just to load the EMCAL geometry.
136  // However, it _is_ necessary for loading the misalignment matrices as of September 2020
137  // Eventually, those matrices will be moved to the CCDB, but it's not yet ready.
138  // FIXME: Hardcoded for run 2
139  o2::base::GeometryManager::loadGeometry(); // for generating full clusters
140  LOG(DEBUG) << "After load geometry!";
141  o2::emcal::Geometry* geometry = o2::emcal::Geometry::GetInstanceFromRunNumber(223409);
142  if (!geometry) {
143  LOG(ERROR) << "Failure accessing geometry";
144  }
145 
146  JetFinderTaskBase::Init();
147  }
148 
149  void process(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels>>::iterator const& collision,
150  soa::Filtered<soa::Join<aod::Tracks, aod::TrackSelection>> const& tracks,
151  aod::EMCALClusters const & clusters)
152  {
153  if (!collision.alias()[kINT7]) {
154  return; //remove hard code
155  }
156  if (!collision.sel7()) {
157  return; //remove hard code
158  }
159  processBegin();
160 
161  for (auto& track : tracks) {
162  fillConstituents(track, inputParticles);
163  inputParticles.back().set_user_index(track.globalIndex());
164  }
165  for (auto & cluster : clusters) {
166  // The right thing to do here would be to fully calculate the momentum depending on the position.
167  // However, it's not clear that this functionality exists yet (21 June 2021)
168  inputParticles.emplace_back(fastjet::PseudoJet());
169  // TODO: This isn't right, but it's the best that can be done at the moment (21 June 2021)
170  inputParticles.back().reset_PtYPhiM(cluster.E(), cluster.Eta(), cluster.Phi(), JetFinder::mPion);
171  inputParticles.back().set_user_index(cluster.globalIndex());
172  }
173 
174  processBase();
175  }
176 };
