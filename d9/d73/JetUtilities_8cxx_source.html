36 template<typename T>
37 std::tuple<std::vector<std::size_t>, std::vector<T>, std::vector<T>> DuplicateJetsAroundPhiBoundary(
38  std::vector<T> & jetsPhi,
39  std::vector<T> & jetsEta,
40  double maxMatchingDistance,
41  // TODO: Remove additional margin after additional testing.
42  double additionalMargin = 0.05
43 )
44 {
45  const std::size_t nJets = jetsPhi.size();
46  std::vector<std::size_t> jetsMapToJetIndex(nJets);
47  // We need to keep track of the map from the duplicated vector to the existing jets.
48  // To start, we fill this map (practically, it maps from vector index to an index, so we
49  // just use a standard vector) with the existing jet indices, which range from 0..nJets.
50  std::iota(jetsMapToJetIndex.begin(), jetsMapToJetIndex.end(), 0);
51 
52  // The full duplicated jets will be stored in a new vector to avoid disturbing the input data.
53  std::vector<T> jetsPhiComparison(jetsPhi);
54  std::vector<T> jetsEtaComparison(jetsEta);
55 
56  // Duplicate the jets
57  // When a jet is outside of the desired phi range, we make a copy of the jet, shifting it by 2pi
58  // as necessary. The eta value is copied directly, as is the index of the original jet, so we can
59  // map from index in duplicated data -> index in original collection.
60  // NOTE: Assumes, but does not validate, that 0 <= phi < 2pi.
61  for (std::size_t i = 0; i < nJets; i++) {
62  // Handle lower edge
63  if (jetsPhi[i] <= (maxMatchingDistance + additionalMargin)) {
64  jetsPhiComparison.emplace_back(jetsPhi[i] + 2 * M_PI);
65  jetsEtaComparison.emplace_back(jetsEta[i]);
66  jetsMapToJetIndex.emplace_back(jetsMapToJetIndex[i]);
67  }
68  // Handle upper edge
69  if (jetsPhi[i] >= (2 * M_PI - (maxMatchingDistance + additionalMargin))) {
70  jetsPhiComparison.emplace_back(jetsPhi[i] - 2 * M_PI);
71  jetsEtaComparison.emplace_back(jetsEta[i]);
72  jetsMapToJetIndex.emplace_back(jetsMapToJetIndex[i]);
73  }
74  }
75 
76  return std::move(std::make_tuple(jetsMapToJetIndex, jetsPhiComparison, jetsEtaComparison));
77 }

79 template<typename T>
80 std::tuple<std::vector<int>, std::vector<int>> MatchJetsGeometricallyImpl(
81  const std::vector<T> & jetsBasePhi,
82  const std::vector<T> & jetsBaseEta,
83  std::vector<T> jetsBasePhiForMatching,
84  std::vector<T> jetsBaseEtaForMatching,
85  const std::vector<std::size_t> jetMapBaseToJetIndex,
86  const std::vector<T> & jetsTagPhi,
87  const std::vector<T> & jetsTagEta,
88  std::vector<T> jetsTagPhiForMatching,
89  std::vector<T> jetsTagEtaForMatching,
90  const std::vector<std::size_t> jetMapTagToJetIndex,
91  double maxMatchingDistance
92  )
93 {
94  // Validation
95  // If no jets in either collection, then return immediately.
96  const std::size_t nJetsBase = jetsBaseEta.size();
97  const std::size_t nJetsTag = jetsTagEta.size();
98  if(!(nJetsBase && nJetsTag)) {
99  return std::make_tuple(std::vector<int>(nJetsBase), std::vector<int>(nJetsTag));
100  }
101  // Require that the comparison vectors are greater than or equal to the standard collections.
102  if (jetsBasePhiForMatching.size() < jetsBasePhi.size()) {
103  throw std::invalid_argument("Base collection phi for matching is smaller than the input base collection.");
104  }
105  if (jetsBaseEtaForMatching.size() < jetsBaseEta.size()) {
106  throw std::invalid_argument("Base collection eta for matching is smaller than the input base collection.");
107  }
108  if (jetsTagPhiForMatching.size() < jetsTagPhi.size()) {
109  throw std::invalid_argument("Tag collection phi for matching is smaller than the input tag collection.");
110  }
111  if (jetsTagEtaForMatching.size() < jetsTagEta.size()) {
112  throw std::invalid_argument("Tag collection eta for matching is smaller than the input tag collection.");
113  }
114 
115  // Build the KD-trees using vectors
116  // We build two trees:
117  // treeBase, which contains the base collection.
118  // treeTag, which contains the tag collection.
119  // The trees are built to match in two dimensions (eta, phi)
120  TKDTree<int, T> treeBase(jetsBaseEtaForMatching.size(), 2, 1), treeTag(jetsTagEtaForMatching.size(), 2, 1);
121  // By utilizing SetData, we can avoid having to copy the data again.
122  treeBase.SetData(0, jetsBaseEtaForMatching.data());
123  treeBase.SetData(1, jetsBasePhiForMatching.data());
124  treeBase.Build();
125  treeTag.SetData(0, jetsTagEtaForMatching.data());
126  treeTag.SetData(1, jetsTagPhiForMatching.data());
127  treeTag.Build();
128 
129  // Storage for the jet matching indices.
130  // matchIndexTag maps from the base index to the tag index.
131  // matchBaseTag maps from the tag index to the base index.
132  std::vector<int> matchIndexTag(nJetsBase, -1), matchIndexBase(nJetsTag, -1);
133 
134  // Find the tag jet closest to each base jet.
135  for (std::size_t iBase = 0; iBase < nJetsBase; iBase++) {
136  Double_t point[2] = {jetsBaseEta[iBase], jetsBasePhi[iBase]};
137  Int_t index(-1); Double_t distance(-1);
138  treeTag.FindNearestNeighbors(point, 1, &index, &distance);
139  // test whether indices are matching:
140  if (index >= 0 && distance < maxMatchingDistance) {
141  LOG(DEBUG) << "Found closest tag jet for " << iBase << " with match index " << index << " and distance " << distance << "\n";
142  matchIndexTag[iBase] = index;
143  } else {
144  LOG(DEBUG) << "Closest tag jet not found for " << iBase << ", distance to closest " << distance << "\n";
145  }
146  }
147 
148  // Find the base jet closest to each tag jet
149  for (std::size_t iTag = 0; iTag < nJetsTag; iTag++) {
150  Double_t point[2] = {jetsTagEta[iTag], jetsTagPhi[iTag]};
151  Int_t index(-1); Double_t distance(-1);
152  treeBase.FindNearestNeighbors(point, 1, &index, &distance);
153  if(index >= 0 && distance < maxMatchingDistance) {
154  LOG(DEBUG) << "Found closest base jet for " << iTag << " with match index " << index << " and distance " << distance << std::endl;
155  matchIndexBase[iTag] = index;
156  } else {
157  LOG(DEBUG) << "Closest tag jet not found for " << iTag << ", distance to closest " << distance << "\n";
158  }
159  }
160 
161  // Convert indices in the duplicated jet vectors into the original jet indices.
162  // First for the base -> tag map.
163  for (auto & v : matchIndexTag) {
164  // If it's -1, it means that it didn't find a matching jet.
165  // We have to explicitly check for it here because it would be an invalid index.
166  if (v != -1) {
167  v = jetMapTagToJetIndex[v];
168  }
169  }
170  // Then for the index -> base map.
171  for (auto & v : matchIndexBase) {
172  if (v != -1) {
173  v = jetMapBaseToJetIndex[v];
174  }
175  }
176 
177  // Finally, we'll check for true matches, which are pairs where the base jet is the
178  // closest to the tag jet and vice versa
179  // As the lists are linear a loop over the outer base jet is sufficient.
180  std::vector<int> baseToTagMap(nJetsBase, -1);
181  std::vector<int> tagToBaseMap(nJetsTag, -1);
182  LOG(DEBUG) << "Starting true jet loop: nbase(" << nJetsBase << "), ntag(" << nJetsTag << ")\n";
183  for (std::size_t iBase = 0; iBase < nJetsBase; iBase++) {
184  LOG(DEBUG) << "base jet " << iBase << ": match index in tag jet container " << matchIndexTag[iBase] << "\n";
185  if (matchIndexTag[iBase] > -1){
186  LOG(DEBUG) << "tag jet " << matchIndexTag[iBase] << ": matched base jet " << matchIndexBase[matchIndexTag[iBase]] << "\n";
187  }
188  if (matchIndexTag[iBase] > -1 && matchIndexBase[matchIndexTag[iBase]] == iBase) {
189  LOG(DEBUG) << "True match! base index: " << iBase << ", tag index: " << matchIndexTag[iBase] << "\n";
190  baseToTagMap[iBase] = matchIndexTag[iBase];
191  tagToBaseMap[matchIndexTag[iBase]] = iBase;
192  }
193  }
194 
195  return std::make_tuple(baseToTagMap, tagToBaseMap);
196 }

198 template<typename T>
199 std::tuple<std::vector<int>, std::vector<int>> MatchJetsGeometrically(
200  std::vector<T> jetsBasePhi,
201  std::vector<T> jetsBaseEta,
202  std::vector<T> jetsTagPhi,
203  std::vector<T> jetsTagEta,
204  double maxMatchingDistance
205 )
206 {
207  // Validation
208  const std::size_t nJetsBase = jetsBaseEta.size();
209  const std::size_t nJetsTag = jetsTagEta.size();
210  if(!(nJetsBase && nJetsTag)) {
211  // There are no jets, so nothing to be done.
212  return std::make_tuple(std::vector<int>(nJetsBase), std::vector<int>(nJetsTag));
213  }
214  // Input sizes must match
215  if (jetsBasePhi.size() != jetsBaseEta.size()) {
216  throw std::invalid_argument("Base collection eta and phi sizes don't match. Check the inputs.");
217  }
218  if (jetsTagPhi.size() != jetsTagEta.size()) {
219  throw std::invalid_argument("Tag collection eta and phi sizes don't match. Check the inputs.");
220  }
221 
222  // To perform matching with periodic boundary conditions (ie. phi) with a KDTree, we need
223  // to duplicate data up to maxMatchingDistance in phi because phi is periodic.
224  // NOTE: vectors are modified in place to avoid copies.
225  auto && [jetMapBaseToJetIndex, jetsBasePhiComparison, jetsBaseEtaComparison] = DuplicateJetsAroundPhiBoundary(jetsBasePhi, jetsBaseEta, maxMatchingDistance);
226  auto && [jetMapTagToJetIndex, jetsTagPhiComparison, jetsTagEtaComparison] = DuplicateJetsAroundPhiBoundary(jetsTagPhi, jetsTagEta, maxMatchingDistance);
227 
228  // Finally, perform the actual matching.
229  auto && [baseToTagMap, tagToBaseMap] = MatchJetsGeometricallyImpl(
230  jetsBasePhi, jetsBaseEta, jetsBasePhiComparison, jetsBaseEtaComparison, jetMapBaseToJetIndex,
231  jetsTagPhi, jetsTagEta, jetsTagPhiComparison, jetsTagEtaComparison, jetMapTagToJetIndex,
232  maxMatchingDistance
233  );
234 
235  return std::make_tuple(baseToTagMap, tagToBaseMap);
236 }
